```bash
npm install --save @lxsmnsyc/react-use-websockets
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from '@lxsmnsyc/react-use-websockets'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
